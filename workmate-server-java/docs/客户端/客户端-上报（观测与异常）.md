## 一、设计思路

客户端内聚一个 `ObservabilityService`，收集事件（用户行为 + 异常），定时批量 POST 到服务端 `POST /workmate/observability/events`，并集成全局异常捕获。请求经过 EIP 网关，网关自动注入 `X-EIPGW-USERID`。

### 架构图

```
┌─────────────────────────────────────────────────────────┐
│                 ObservabilityService                     │
├─────────────────────────────────────────────────────────┤
│  面包屑 (breadcrumbs: Breadcrumb[], 最近 20 条)         │
│       │  addBreadcrumb() 入队                            │
│       ▼  reportErrorEvent() 自动附加                     │
│                                                          │
│  内存队列 (eventQueue: ObservabilityEventItem[])        │
│       │  每事件携带 eventId (UUID) 用于去重              │
│       ▼                                                  │
│  ┌──────────────┐   失败    ┌──────────────────────┐    │
│  │ 定时冲刷      │ ────────→ │ 本地磁盘缓存          │    │
│  │ (5s 间隔)     │           │ ~/.proma/            │    │
│  │ 最多 50 条/批 │ ←──────── │ obs-queue.jsonl      │    │
│  └──────┬───────┘   恢复     └──────────────────────┘    │
│         │ 成功                                            │
│         ▼                                                 │
│  ┌──────────────────────────────────────────────────┐   │
│  │ POST {serverUrl}/workmate/observability/events   │   │
│  │ Headers:                                          │   │
│  │   Content-Type: application/json                  │   │
│  │   Cookie: EIPGW-TOKEN=<jwt>                      │   │
│  │ (EIP 网关自动注入 X-EIPGW-USERID)                  │   │
│  │ Body: { events: [...] }                           │   │
│  └──────────────────────────────────────────────────┘   │
└─────────────────────────────────────────────────────────┘
```

> 采样/限流层已从架构图中移除（流量低，不需要）。`ObservabilityConfig.sampleRate` / `maxEventsPerMinute` 字段保留。

## 二、数据模型

类型文件：`apps/electron/src/types/workmate.ts`（统一在 workmate 命名空间下，不侵入 `packages/shared/src/types/index.ts`）。

```typescript
export type ObservabilityEventType =
  | 'user_login' | 'user_logout' | 'chat_question' | 'agent_question'
  | 'error' | 'upgrade_check'

export interface ObservabilityEventItem {
  eventId: string            // 事件唯一标识（UUID），用于服务端去重
  type: ObservabilityEventType
  userId: string             // 用户工号（由 auth-service.getJobId() 填充，服务端以 req.jobId 覆盖）
  timestamp: number
  question?: string
  questionLength?: number
  modelId?: string
  channelId?: string
  sessionId?: string
  workspaceId?: string
  result?: 'success' | 'failure' | 'pending'
  responseDurationMs?: number
  error?: { type: string; message: string; stack?: string; statusCode?: number; fingerprint?: string }
  breadcrumbs?: Breadcrumb[] // 事件发生前的用户操作路径（error 事件自动附加）
  /**
   * 扩展标签。用于 reportErrorEvent 传入 context 参数。
   * 类型限定为 string→string，避免 spread 任意键污染 ObservabilityEventItem。
   */
  tags?: Record<string, string>
  client: { appVersion: string; platform: string; osVersion: string }
}

export interface Breadcrumb {
  type: 'navigation' | 'user_action' | 'system' | 'http'
  category: string
  message: string
  timestamp: number
  /** 受控大小（仅允许基础类型），单条面包屑 data 序列化后不超过 4KB */
  data?: Record<string, string | number | boolean | null>
}

export interface ObservabilityReportBody {
  events: ObservabilityEventItem[]
}

export interface ObservabilityConfig {
  enabled: boolean
  url: string              // 完整上报地址，如 https://host/workmate/observability/events
  timeoutMs?: number
  maxQueueSize?: number
  flushIntervalMs?: number
  maxBatchSize?: number
  /** 采样率 0.0~1.0，默认 1.0（暂不实现，字段保留） */
  sampleRate?: number
  /** 每分钟最大事件数，默认 60（暂不实现，字段保留） */
  maxEventsPerMinute?: number
  /** 是否启用面包屑，默认 true */
  enableBreadcrumbs?: boolean
  /** 面包屑最大数量，默认 20 */
  maxBreadcrumbs?: number
  /** 单个事件序列化后最大字节数，默认 256KB（防面包屑 data 过大撑爆 payload） */
  maxEventBytes?: number
}
```

### 3.1 事件入队核心逻辑

```typescript
// observability-service.ts 核心实现概要

import { createHash, randomUUID } from 'node:crypto'
import { existsSync, mkdirSync, readFileSync, appendFileSync, writeFileSync } from 'node:fs'
import { join } from 'node:path'
import { app } from 'electron'
import { getToken, getJobId } from './auth-service'
import { getConfigDir } from './config-paths'
import { getWorkmateConfig } from './workmate-config-service'
import { safeStringify } from './utils/safe-stringify'

const breadcrumbs: Breadcrumb[] = []
let eventQueue: ObservabilityEventItem[] = []
let diskCachePath: string | null = null
let config: ObservabilityConfig | null = null
let flushTimer: ReturnType<typeof setInterval> | null = null
let isFlushing = false
let isReportingError = false

// ===== 初始化 =====

export function init(obsConfig: ObservabilityConfig): void {
  config = obsConfig
  if (!config.enabled) return

  // 磁盘缓存路径
  const dir = getConfigDir()
  if (!existsSync(dir)) mkdirSync(dir, { recursive: true })
  diskCachePath = join(dir, 'obs-queue.jsonl')

  // 启动时从磁盘恢复未发送的事件
  restoreFromDiskCache()

  // 启动定时冲刷
  flushTimer = setInterval(() => {
    void flushQueue()
  }, config.flushIntervalMs ?? 5000)
}

// ===== 关闭 =====

export function shutdown(): void {
  if (flushTimer) {
    clearInterval(flushTimer)
    flushTimer = null
  }
  // 同步冲刷：最后再尝试一次
  void flushQueue()
}

// ===== 面包屑记录 =====

export function addBreadcrumb(crumb: Omit<Breadcrumb, 'timestamp'>): void {
  if (!config?.enableBreadcrumbs) return
  const max = config.maxBreadcrumbs ?? 20
  breadcrumbs.push({ ...crumb, timestamp: Date.now() } as Breadcrumb)
  while (breadcrumbs.length > max) {
    breadcrumbs.shift()
  }
}

// ===== 通用事件入队 =====

export function reportEvent(item: Omit<ObservabilityEventItem, 'eventId'>): void {
  if (!config?.enabled) return

  const event: ObservabilityEventItem = {
    ...item,
    eventId: randomUUID(),
  }

  // 单事件体积保护：序列化后超过 maxEventBytes 拒绝入队
  const maxBytes = config.maxEventBytes ?? 256 * 1024
  if (safeStringify(event).length > maxBytes) {
    console.warn('[观测上报] 事件体积超限，已丢弃', { type: event.type, bytes: safeStringify(event).length })
    return
  }

  eventQueue.push(event)
  if (eventQueue.length >= (config.maxQueueSize ?? 200)) {
    void flushQueue()
  }
}

// ===== Chat 提问上报 =====

export function reportChatEvent(params: {
  userId: string; question: string; modelId: string;
  result: 'success' | 'failure'; responseDurationMs: number; error?: Error;
}): void {
  reportEvent({
    type: 'chat_question',
    userId: params.userId,
    timestamp: Date.now(),
    question: params.question,
    questionLength: params.question.length,
    modelId: params.modelId,
    result: params.result,
    responseDurationMs: params.responseDurationMs,
    error: params.error ? {
      type: params.error.name,
      message: params.error.message,
      stack: params.error.stack,
    } : undefined,
    client: getClientInfo(),
  })
}

// ===== 登录事件上报 =====

export function reportLoginEvent(
  jobId: string, result: 'success' | 'failure', error?: Error,
): void {
  reportEvent({
    type: 'user_login',
    userId: jobId,
    timestamp: Date.now(),
    result,
    error: error ? { type: error.name, message: error.message } : undefined,
    client: getClientInfo(),
  })
}

// ===== 冲刷上报（修复：失败时回写队列 + 落盘） =====

export async function flushQueue(): Promise<void> {
  if (isFlushing) return  // 防并发：同一时刻只有一个 flush
  if (eventQueue.length === 0) return
  if (!config?.enabled || !config.url) return

  isFlushing = true
  const batch = eventQueue.splice(0, config.maxBatchSize ?? 50)
  const token = await getToken()

  try {
    const response = await fetch(config.url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        ...(token ? { 'Cookie': `EIPGW-TOKEN=${token}` } : {}),
      },
      body: JSON.stringify({ events: batch }),
      signal: AbortSignal.timeout(config.timeoutMs ?? 5000),
    })

    if (!response.ok) {
      throw new Error(`HTTP ${response.status}`)
    }

    // 业务码校验：服务端返回 { code, message, data }，仅 code === 0 视为成功
    // 否则（如 code === 1 业务错误）也走失败重试路径，避免事件被静默丢弃
    const body = await response.json().catch(() => null) as { code?: number; message?: string } | null
    if (!body || body.code !== 0) {
      throw new Error(`业务错误 code=${body?.code ?? 'unknown'} message=${body?.message ?? ''}`)
    }
  } catch (error) {
    // 失败时回写到队列头部（保持事件顺序），同时追加到磁盘
    eventQueue.unshift(...batch)
    appendToDiskCache(batch)
    console.warn('[观测上报] 上报失败，已回写队列并落盘:', error)
  } finally {
    isFlushing = false
  }
}

// ===== 磁盘缓存（失败兜底） =====

function appendToDiskCache(batch: ObservabilityEventItem[]): void {
  if (!diskCachePath) return
  try {
    const lines = batch.map((e) => safeStringify(e)).join('\n') + '\n'
    appendFileSync(diskCachePath, lines, 'utf-8')
  } catch (error) {
    console.warn('[观测上报] 写磁盘缓存失败:', error)
  }
}

function restoreFromDiskCache(): void {
  if (!diskCachePath || !existsSync(diskCachePath)) return
  try {
    const content = readFileSync(diskCachePath, 'utf-8')
    const lines = content.split('\n').filter(Boolean)
    const restored: ObservabilityEventItem[] = []
    for (const line of lines) {
      try {
        restored.push(JSON.parse(line) as ObservabilityEventItem)
      } catch {
        // 单行损坏忽略
      }
    }
    if (restored.length > 0) {
      eventQueue.unshift(...restored)
      // 恢复后清空磁盘文件（事件已回到内存队列）
      writeFileSync(diskCachePath, '', 'utf-8')
      console.log(`[观测上报] 从磁盘恢复 ${restored.length} 条事件`)
    }
  } catch (error) {
    console.warn('[观测上报] 读取磁盘缓存失败:', error)
  }
}
```

## 四、异常捕获与上报

### 4.1 事件监听清单（确保不漏不重）

| 监听点 | 触发场景 | 监听方式 | 上报 API |
|--------|---------|---------|---------|
| 主进程未捕获异常 | Node.js 主进程抛出未处理异常 | `process.on('uncaughtException')` | `reportErrorEvent(err, { tags: { source: 'uncaughtException' } })` |
| 主进程未处理 Promise 拒绝 | 主进程 async 函数 reject 未 catch | `process.on('unhandledRejection')` | `reportErrorEvent(err, { tags: { source: 'unhandledRejection' } })` |
| 主进程 Electron 渲染进程崩溃 | GPU 进程、utility process 崩溃 | `app.on('render-process-gone')` | `reportErrorEvent(err, { tags: { source: 'render-process-gone' } })` |
| 渲染进程 React 错误边界 | 组件渲染时抛错 | 扩展 `TabErrorBoundary.componentDidCatch` | 通过 IPC 通知主进程 → `reportErrorEvent` |
| Chat 请求失败 | `chat-service.ts` catch 块 | **不直接修改 chat-service**，由 IPC 错误事件触发 | 监听 `CHAT_IPC_CHANNELS.STREAM_ERROR` 事件 |
| Agent 请求失败 | `agent-orchestrator.ts` catch 块 | **不直接修改 agent-orchestrator**，由 agent-event-bus 触发 | 订阅 `AgentEventBus` 中 `error` 事件 |
| 升级检测失败 | `upgrade-check-service.ts` catch 块 | **不直接修改**，在新模块内 catch 后调用 | `reportErrorEvent` |

> **"不漏不重"原则**：
> - **不漏**：每种错误源必须有且仅有一个监听点（主进程 / 渲染进程分别独立监听）。
> - **不重**：单一错误不被多个监听器重复上报。`reportErrorEvent` 内部的 `isReportingError` 标志防递归；`fingerprint` 字段用于服务端去重。

### 4.2 防递归保护

```typescript
let isReportingError = false

export function reportErrorEvent(error: Error, context?: { tags?: Record<string, string> }): void {
  if (isReportingError) return
  if (!config?.enabled) return
  isReportingError = true
  try {
    const fingerprint = generateFingerprint(error.name, error.message)
    reportEvent({
      type: 'error',
      userId: getJobId() ?? 'unknown',
      timestamp: Date.now(),
      result: 'failure',
      error: {
        type: error.name,
        message: error.message,
        stack: error.stack,
        fingerprint,
      },
      breadcrumbs: [...breadcrumbs],
      tags: context?.tags,
      client: getClientInfo(),
    })
  } finally {
    isReportingError = false
  }
}

function generateFingerprint(errorType: string, errorMessage: string): string {
  const raw = `${errorType}:${errorMessage.slice(0, 100)}`
  return createHash('md5').update(raw).digest('hex').slice(0, 16)
}
```

> `context` 类型从 `Record<string, unknown>` 收紧为 `{ tags?: Record<string, string> }`，避免 spread 任意键污染事件结构。

### 4.3 主进程全局异常捕获

新增文件：`apps/electron/src/main/lib/error-handler.ts`

```typescript
import { reportErrorEvent } from './observability-service'
import { app, type WebContents } from 'electron'

/**
 * 致命错误判定：以下错误类型视为不可恢复，需上报后退出进程
 * - 资源耗尽（ENOMEM）
 * - 关键模块加载失败
 * - Electron 内部错误
 */
const FATAL_ERROR_PATTERNS = [
  /ENOMEM/,
  /JavaScript heap out of memory/,
  /FATAL: out of memory/,
  /Cannot find module '\.\/build\/Release\//,
]

function isFatalError(error: Error): boolean {
  const text = `${error.name} ${error.message} ${error.stack ?? ''}`
  return FATAL_ERROR_PATTERNS.some((re) => re.test(text))
}

export function registerGlobalErrorHandlers(): void {
  // 1) Node.js 未捕获异常
  process.on('uncaughtException', (error) => {
    console.error('[Global Error] 未捕获异常:', error)
    reportErrorEvent(error, { tags: { source: 'uncaughtException' } })
    if (isFatalError(error)) {
      // 致命错误：先同步上报（fire-and-forget），再退出
      setTimeout(() => app.exit(1), 1000)
    }
  })

  // 2) Node.js 未处理 Promise 拒绝
  process.on('unhandledRejection', (reason) => {
    console.error('[Global Error] 未处理的 Promise 拒绝:', reason)
    const error = reason instanceof Error ? reason : new Error(String(reason))
    reportErrorEvent(error, { tags: { source: 'unhandledRejection' } })
    // unhandledRejection 默认不退出进程（避免掩盖），仅上报
  })

  // 3) Electron 渲染进程 / 子进程崩溃
  app.on('render-process-gone', (_event, webContents: WebContents, details: { reason: string; exitCode: number }) => {
    const error = new Error(`渲染进程崩溃: ${details.reason} (exitCode=${details.exitCode})`)
    error.name = 'RenderProcessGoneError'
    reportErrorEvent(error, {
      tags: {
        source: 'render-process-gone',
        reason: details.reason,
        exitCode: String(details.exitCode),
        url: webContents.getURL(),
      },
    })
  })

  // 4) 子进程崩溃
  app.on('child-process-gone', (_event, details: { type: string; reason: string; exitCode: number }) => {
    const error = new Error(`子进程崩溃: ${details.type} ${details.reason} (exitCode=${details.exitCode})`)
    error.name = 'ChildProcessGoneError'
    reportErrorEvent(error, {
      tags: {
        source: 'child-process-gone',
        type: details.type,
        reason: details.reason,
        exitCode: String(details.exitCode),
      },
    })
  })
}
```

  name: string
  message: string
  stack?: string
  componentStack?: string

### 4.6 Chat / Agent 错误接入（不修改核心文件）

监听 IPC 错误事件（不修改 `chat-service.ts` / `agent-orchestrator.ts`）：

```typescript
// 在 workmate-init.ts 中
import { CHAT_IPC_CHANNELS } from '@proma/shared'
import { reportErrorEvent } from './observability-service'

export function initWorkmateServices(): void {
  // ... 其他初始化

  // 监听 Chat 流式错误事件
  // 注：chat-service 在 stream-error 时通过 webContents.send 发送 CHAT_IPC_CHANNELS.STREAM_ERROR
  // 我们订阅该事件做错误上报（不影响原有渲染逻辑）
  // 由于 ipcMain 不直接接收 webContents.send 事件，这里采用"渲染进程转发"模式：
  //   渲染进程 → IPC 主动通知主进程 → 上报
  ipcMain.on(WORKMATE_IPC_CHANNELS.CHAT_ERROR_REPORT, (_event, payload) => {
    const error = new Error(payload.message)
    error.name = payload.type ?? 'ChatStreamError'
    error.stack = payload.stack
    reportErrorEvent(error, {
      tags: {
        source: 'chat-stream',
        sessionId: payload.sessionId,
        statusCode: String(payload.statusCode ?? ''),
      },
    })
  })

  // 监听 Agent 错误事件
  ipcMain.on(WORKMATE_IPC_CHANNELS.AGENT_ERROR_REPORT, (_event, payload) => {
    const error = new Error(payload.message)
    error.name = payload.type ?? 'AgentError'
    error.stack = payload.stack
    reportErrorEvent(error, {
      tags: {
        source: 'agent',
        sessionId: payload.sessionId,
      },
    })
  })

  // 监听升级检测错误
  ipcMain.on(WORKMATE_IPC_CHANNELS.UPGRADE_CHECK_ERROR, (_event, payload) => {
    const error = new Error(payload.message)
    error.name = 'UpgradeCheckError'
    reportErrorEvent(error, { tags: { source: 'upgrade-check' } })
  })
}
```

> 渲染进程在 `chat-service` / `agent-orchestrator` 的错误回调中**只需一行** `window.electronAPI.reportChatError({...})` 即可，无需修改任何 Proma 核心文件。

### 4.7 面包屑集成

在关键交互点记录面包屑，异常上报时自动附加：

| 集成点 | 面包屑示例 |
|--------|-----------|
| 路由切换 | `{ type: 'navigation', category: 'route', message: '进入设置页' }` |
| 按钮点击 | `{ type: 'user_action', category: 'ui.click', message: '点击发送按钮' }` |
| API 调用 | `{ type: 'http', category: 'api.call', message: 'POST /workmate/observability/events' }` |
| 系统事件 | `{ type: 'system', category: 'app.lifecycle', message: '应用启动' }` |

**大小保护**：
- `maxBreadcrumbs` 默认 20（可配置）
- 单条 `data` 字段类型收紧为 `Record<string, string | number | boolean | null>`，禁止嵌套对象防爆栈
- 单事件序列化后 `maxEventBytes` 默认 256KB（可配置），超限自动丢弃并 warn

## 六、安全注意事项

| 项 | 措施 |
|----|------|
| 身份不可伪造 | 服务端入库时用 `req.jobId` 覆盖客户端上报的 `userId`，确保身份真实 |
| 上报内容脱敏 | 异常堆栈中过滤本地文件路径中的用户名 |
| 防递归上报 | `isReportingError` 标记防止上报异常触发自身递归 |
| 防并发冲刷 | `isFlushing` 标记防止同时多个 flush 协程并发 |
| 上报失败 | fire-and-forget，失败回写队列 + 追加到 `obs-queue.jsonl`，启动时自动恢复 |
| 事件去重 | 每事件携带 `eventId`（UUID），服务端幂等去重 |
| 体积保护 | `maxEventBytes` 限制单事件大小，防止大面包屑撑爆 payload |
| 采样与限流 | 暂不实现，配置字段保留；`fingerprint` 字段为服务端后续做聚合去重提供依据 |
