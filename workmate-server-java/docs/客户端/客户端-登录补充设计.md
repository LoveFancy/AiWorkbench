## 一、需求概述

### 1.1 不强制登录

客户端**不强制登录**。未登录状态下，所有功能正常可用（本地模型配置、Chat、Agent 等），仅在需要平台能力时提示登录。

### 1.2 新增"泰为平台模型"

在 **设置 → 模型配置** 页面中，新增"泰为平台模型"区块，位于"模型配置"和"Agent 供应商"之前，三者并列：

```
┌─────────────────────────────────────────┐
│  设置 → 模型配置                         │
│                                         │
│  ┌─────────────────────────────────┐    │
│  │ 🏢 泰为平台模型          [刷新]  │ ← 新增，最前方  │
│  │   登录后自动获取平台模型和 Key    │    │
│  │   ┌─────────────────────────┐   │    │
│  │   │ ☑ claude-sonnet-4-5    │   │    │
│  │   │ ☐ deepseek-v4-pro      │   │    │
│  │   └─────────────────────────┘   │    │
│  └─────────────────────────────────┘    │
│                                         │
│  ┌─────────────────────────────────┐    │
│  │ 📡 模型配置              [+添加] │    │
│  │   ...                             │    │
│  └─────────────────────────────────┘    │
```

**未登录时**：显示"请登录 OA 后获取平台模型"，附带登录按钮，点击触发登录弹窗。

**已登录时**：自动调用 `GET /workmate/models` 拉取平台模型列表，展示模型名称、启用开关，支持手动刷新。

```typescript
// apps/electron/src/platform-models/platform-model-service.ts

import { safeStorage } from 'electron'
import { readFileSync, writeFileSync, existsSync } from 'node:fs'
import { join } from 'node:path'
import { getJobId } from '../auth'
import { httpGet } from '../shared/hteip-client'
import type { PlatformModelInfo, PlatformModelsResponse, PlatformModelsCache } from './types'

const CACHE_FILE = 'platform-models-cache.json'
const DEFAULT_TTL_MS = 3_600_000  // 1 小时
const PLATFORM_CHANNEL_ID = '__platform__'

let _configDir: string | null = null
let _serverUrl: string | null = null

/** 初始化（由 main/index.ts 调用） */
export function initPlatformModelService(configDir: string, serverUrl?: string): void {
  _configDir = configDir
  _serverUrl = serverUrl ?? null
}

function cachePath(): string {
  if (!_configDir) throw new Error('Platform model service not initialized')
  return join(_configDir, CACHE_FILE)
}

// ===== L1 内存缓存 =====

let l1ApiKey: string | null = null
let l1Models: PlatformModelInfo[] = []
let l1Total: number = 0
let l1CachedAt: number = 0
let l1JobId: string | null = null

// ===== 总入口 =====

export async function fetchPlatformModels(
  forceRefresh = false,
): Promise<PlatformModelsResponse> {
  const currentJobId = getJobId()

  if (!currentJobId || !_serverUrl) {
    return { apiKey: '', models: [], total: 0 }
  }

  // 缓存命中
  if (!forceRefresh && isCacheValid(currentJobId)) {
    return { apiKey: l1ApiKey ?? '', models: l1Models, total: l1Total }
  }

  // 网络请求
  const { ok, data } = await httpGet<{ code: number; data: PlatformModelsResponse }>(
    `${_serverUrl}/workmate/models`,
  )

  if (!ok || !data || data.code !== 0) {
    return fallbackToCache(currentJobId)
  }

  const { apiKey, models, total } = data.data

  // 写 L1
  l1ApiKey = apiKey
  l1Models = models
  l1Total = total
  l1CachedAt = Date.now()
  l1JobId = currentJobId

  // 写 L2
  saveCacheToDisk(apiKey, models, total, Date.now(), currentJobId)

  // 同步到渠道系统
  syncToChannel(models, apiKey)

  return { apiKey, models, total }
}

// ===== 读取 =====

export function getPlatformApiKey(): string | null {
  return l1ApiKey
}

export function getPlatformModels(): PlatformModelInfo[] {
  return l1Models
}

// ===== 缓存管理 =====

export function clearCache(): void {
  l1ApiKey = null
  l1Models = []
  l1Total = 0
  l1CachedAt = 0
  l1JobId = null
  writeFileSync(cachePath(), JSON.stringify({}), 'utf-8')
}

export function loadCacheFromDisk(jobId?: string): PlatformModelsResponse | null {
  const filePath = cachePath()
  if (!existsSync(filePath)) return null

  try {
    const cache: PlatformModelsCache = JSON.parse(readFileSync(filePath, 'utf-8'))
    if (!cache.models?.length) return null
    if (jobId && cache.jobId && cache.jobId !== jobId) return null

    let apiKey = ''
    if (cache.encryptedApiKey && safeStorage.isEncryptionAvailable()) {
      apiKey = safeStorage.decryptString(Buffer.from(cache.encryptedApiKey, 'base64'))
    } else if (cache.apiKey) {
      apiKey = cache.apiKey
    }

    l1ApiKey = apiKey
    l1Models = cache.models
    l1Total = cache.total
    l1CachedAt = cache.cachedAt
    l1JobId = cache.jobId ?? null

    return { apiKey, models: cache.models, total: cache.total }
  } catch {
    return null
  }
}

// ===== 内部工具 =====

function isCacheValid(jobId: string): boolean {
  if (!l1Models.length) return false
  if (Date.now() - l1CachedAt > DEFAULT_TTL_MS) return false
  if (l1JobId && l1JobId !== jobId) return false
  return true
}

function fallbackToCache(jobId: string): PlatformModelsResponse {
  if (l1JobId === jobId && l1Models.length) {
    return { apiKey: l1ApiKey ?? '', models: l1Models, total: l1Total }
  }
  return { apiKey: '', models: [], total: 0 }
}

function saveCacheToDisk(
  apiKey: string, models: PlatformModelInfo[], total: number,
  cachedAt: number, jobId: string,
): void {
  const cache: PlatformModelsCache = { models, total, cachedAt, jobId }
  if (safeStorage.isEncryptionAvailable()) {
    cache.encryptedApiKey = safeStorage.encryptString(apiKey).toString('base64')
  } else {
    cache.apiKey = apiKey
  }
  writeFileSync(cachePath(), JSON.stringify(cache, null, 2), 'utf-8')
}
```

### 4.4 `ipc-handlers.ts`

```typescript
// apps/electron/src/platform-models/ipc-handlers.ts

import { ipcMain } from 'electron'
import { fetchPlatformModels, getPlatformModels, getPlatformApiKey } from './platform-model-service'

export const PLATFORM_MODELS_IPC_CHANNELS = {
  FETCH: 'platform-models:fetch',
  GET_CACHED: 'platform-models:get-cached',
  GET_API_KEY: 'platform-models:get-api-key',
} as const

/**
 * 注册平台模型 IPC handler。
 *
 * 调用方式（main/index.ts 中加一行）：
 *   import { registerPlatformModelsIpcHandlers } from '../platform-models'
 *   registerPlatformModelsIpcHandlers()
 */
export function registerPlatformModelsIpcHandlers(): void {
  ipcMain.handle(PLATFORM_MODELS_IPC_CHANNELS.FETCH, async (_event, forceRefresh?: boolean) => {
    return fetchPlatformModels(forceRefresh)
  })

  ipcMain.handle(PLATFORM_MODELS_IPC_CHANNELS.GET_CACHED, () => {
    return {
      models: getPlatformModels(),
      apiKey: getPlatformApiKey(),
    }
  })

  ipcMain.handle(PLATFORM_MODELS_IPC_CHANNELS.GET_API_KEY, () => {
    return getPlatformApiKey()
  })
}
```

```typescript
import { ipcRenderer } from 'electron'

  return {
    platformModels: {
      fetchModels: (forceRefresh?) =>
        ipcRenderer.invoke(PLATFORM_MODELS_IPC_CHANNELS.FETCH, forceRefresh),
      getCachedModels: () =>
        ipcRenderer.invoke(PLATFORM_MODELS_IPC_CHANNELS.GET_CACHED),
      getApiKey: () =>
        ipcRenderer.invoke(PLATFORM_MODELS_IPC_CHANNELS.GET_API_KEY),
    },
  }
}
```

```typescript
// apps/electron/src/platform-models/PlatformModelsSection.tsx

import React from 'react'
import { useAtomValue, useSetAtom } from 'jotai'
import { RefreshCw, LogIn, CheckCircle2 } from 'lucide-react'
import { Button } from '@/components/ui/button'
import { Switch } from '@/components/ui/switch'
import { SettingsSection, SettingsCard, SettingsRow } from '@/components/settings/primitives'
import { authStateAtom, loginDialogOpenAtom } from '@/auth/renderer'
import {
  platformModelsAtom,
  platformApiKeyAtom,
  platformModelsLoadingAtom,
  platformModelsLastFetchAtom,
} from './atoms'
import type { PlatformModelInfo } from './types'

export function PlatformModelsSection(): React.ReactElement {
  const authState = useAtomValue(authStateAtom)
  const setLoginDialogOpen = useSetAtom(loginDialogOpenAtom)
  const [models, setModels] = useAtom(platformModelsAtom)
  const [apiKey, setApiKey] = useAtom(platformApiKeyAtom)
  const [loading, setLoading] = useAtom(platformModelsLoadingAtom)
  const [lastFetch, setLastFetch] = useAtom(platformModelsLastFetchAtom)

  const isLoggedIn = authState.isLoggedIn

  const handleFetch = React.useCallback(async () => {
    setLoading(true)
    try {
      const result = await (window.electronAPI as any).platformModels.fetchModels(true)
      setModels(result.models)
      setApiKey(result.apiKey || null)
      setLastFetch(Date.now())
    } catch {
      // 静默
    } finally {
      setLoading(false)
    }
  }, [setModels, setApiKey, setLastFetch, setLoading])

  // 登录后自动拉取
  React.useEffect(() => {
    if (isLoggedIn && models.length === 0) {
      handleFetch()
    }
  }, [isLoggedIn])

  // 退出登录后清空
  React.useEffect(() => {
    if (!isLoggedIn) {
      setModels([])
      setApiKey(null)
      setLastFetch(0)
    }
  }, [isLoggedIn])

  const handleToggleModel = React.useCallback((modelId: string, enabled: boolean) => {
    setModels((prev) =>
      prev.map((m) => (m.id === modelId ? { ...m, enabled } : m)),
    )
    // TODO: 持久化用户启用状态
  }, [setModels])

  const lastFetchLabel = lastFetch
    ? `上次更新: ${new Date(lastFetch).toLocaleTimeString()}`
    : ''

  return (
    <SettingsSection
      title="泰为平台模型"
      description="登录 OA 后自动获取您在泰为大模型平台申请的模型和 API Key"
      action={
        isLoggedIn ? (
          <Button size="sm" variant="outline" onClick={handleFetch} disabled={loading}>
            <RefreshCw size={14} className={loading ? 'animate-spin' : ''} />
            <span className="ml-1">刷新</span>
          </Button>
        ) : undefined
      }
    >
      {!isLoggedIn ? (
        /* 未登录状态 */
        <SettingsCard divided={false}>
          <div className="flex flex-col items-center gap-3 py-8">
            <p className="text-sm text-muted-foreground">
              请登录 OA 后获取平台模型
            </p>
            <Button size="sm" onClick={() => setLoginDialogOpen(true)}>
              <LogIn size={14} />
              <span className="ml-1">登录 OA</span>
            </Button>
          </div>
        </SettingsCard>
      ) : loading ? (
        /* 加载中 */
        <SettingsCard divided={false}>
          <div className="flex items-center justify-center gap-2 py-8 text-sm text-muted-foreground">
            <RefreshCw size={14} className="animate-spin" />
            正在获取平台模型...
          </div>
        </SettingsCard>
      ) : models.length === 0 ? (
        /* 已登录但无模型 */
        <SettingsCard divided={false}>
          <div className="flex flex-col items-center gap-3 py-8">
            <p className="text-sm text-muted-foreground">
              暂未获取到平台模型，请确认已在泰为大模型平台申请模型
            </p>
            <Button size="sm" variant="outline" onClick={handleFetch}>
              <RefreshCw size={14} />
              <span className="ml-1">重新获取</span>
            </Button>
          </div>
        </SettingsCard>
      ) : (
        /* 模型列表 */
        <SettingsCard>
          {models.map((model) => (
            <PlatformModelRow
              key={model.id}
              model={model}
              onToggle={(enabled) => handleToggleModel(model.id, enabled)}
            />
          ))}
        </SettingsCard>
      )}

      {/* 状态栏 */}
      {isLoggedIn && apiKey && (
        <div className="flex items-center gap-2 mt-2 px-1">
          <CheckCircle2 size={12} className="text-green-500" />
          <span className="text-xs text-muted-foreground">
            API Key 已获取 · {lastFetchLabel}
          </span>
        </div>
      )}
    </SettingsSection>
  )
}

// ===== 模型行子组件 =====

function PlatformModelRow({
  model,
  onToggle,
}: {
  model: PlatformModelInfo
  onToggle: (enabled: boolean) => void
}): React.ReactElement {
  const isAnthropicCompatible =
    model.provider === 'anthropic' ||
    model.provider?.startsWith('huatai')

  const description = [
    model.provider ? model.provider : undefined,
    model.description || undefined,
    isAnthropicCompatible ? '可用于 Agent' : undefined,
  ]
    .filter(Boolean)
    .join(' · ')

  return (
    <SettingsRow
      label={model.name}
      description={description || undefined}
    >
      <Switch
        checked={model.enabled}
        onCheckedChange={onToggle}
      />
    </SettingsRow>
  )
}
```

## 六、与现有渠道系统的兼容

### 6.1 虚拟渠道 `__platform__`

平台模型拉取成功后，自动在 `channelsAtom` 中创建/更新一个虚拟渠道：

```typescript
const PLATFORM_CHANNEL: Channel = {
  id: '__platform__',
  name: '泰为平台模型',
  provider: 'anthropic',  // 或根据模型协议动态判断
  baseUrl: '',            // 使用默认
  apiKey: platformApiKey, // safeStorage 加密
  models: platformModels.map(m => ({
    id: m.id,
    name: m.name,
    enabled: m.enabled,
  })),
  enabled: true,
}
```

### 6.2 在 ChannelSettings 中的表现

| 区块 | 虚拟渠道表现 |
|------|-------------|
| 泰为平台模型 | 专属 UI，显示模型列表 + 开关 |
| 模型配置 | 显示为一条特殊渠道，不可编辑/删除 |
| Agent 供应商 | 若协议为 Anthropic 兼容，显示开关 |

### 6.3 Agent 兼容性判断

平台模型是否可用于 Agent，取决于其 `provider` 字段：

| provider 值 | Agent 兼容 |
|-------------|-----------|
| `anthropic` | 是 |
| `huatai-anthropic` | 是 |
| `openai` | 否 |
| 其他 | 否 |

判断逻辑复用 `@proma/shared` 中的 `isAgentCompatibleProvider()`。

## 八、注意事项

1. **不强制登录**：`App.tsx` 中不插入 `ForceLoginGuard`，用户可跳过登录直接使用本地功能
2. **回合上游**：所有新增代码在 `platform-models/` 独立目录内，修改均为接线式，冲突风险极低
3. **登录弹窗复用**：平台模型区块的"登录 OA"按钮直接设置 `loginDialogOpenAtom = true`，复用 `App.tsx` 中已有的 `<LoginDialog>` 组件
4. **退出登录联动**：`authStateAtom.isLoggedIn` 变为 `false` 时，`PlatformModelsSection` 自动清空模型列表
5. **API Key 安全**：与 `auth.json` 一致，使用 `safeStorage` 加密落盘
6. **缓存策略**：默认 TTL 1 小时，与 `models-cache.json` 独立，存储在 `platform-models-cache.json`
