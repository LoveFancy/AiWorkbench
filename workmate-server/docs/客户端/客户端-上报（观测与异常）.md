## 一、设计思路

客户端内聚一个 `ObservabilityService`，收集事件（用户行为 + 异常），定时批量 POST 到服务端 `POST /workmate/observability/events`，并集成全局异常捕获。请求经过 EIP 网关，网关自动注入 `X-EIPGW-USERID`。

### 架构图

```
┌─────────────────────────────────────────────────────────┐
│                 ObservabilityService                     │
├─────────────────────────────────────────────────────────┤
│  面包屑 (breadcrumbs: Breadcrumb[], 最近 20 条)         │
│       │  addBreadcrumb() 入队                            │
│       ▼  reportErrorEvent() 自动附加                     │
│                                                          │
│  采样 & 限流层                                           │
│  ┌──────────────────────────────────────────────────┐   │
│  │ error 事件 → 直通（全量上报）                       │   │
│  │ 其他事件 → 采样率检查 + 每分钟限流                   │   │
│  └──────────────────────────────────────────────────┘   │
│       │ 通过                                             │
│       ▼                                                  │
│  内存队列 (eventQueue: ObservabilityEventItem[])        │
│       │  每事件携带 eventId (UUID) 用于去重              │
│       ▼                                                  │
│  ┌──────────────┐   失败    ┌──────────────────────┐    │
│  │ 定时冲刷      │ ────────→ │ 本地磁盘缓存          │    │
│  │ (5s 间隔)     │           │ ~/.proma/            │    │
│  │ 最多 50 条/批 │ ←──────── │ obs-queue.jsonl      │    │
│  └──────┬───────┘   恢复     └──────────────────────┘    │
│         │ 成功                                            │
│         ▼                                                 │
│  ┌──────────────────────────────────────────────────┐   │
│  │ POST {serverUrl}/workmate/observability/events   │   │
│  │ Headers:                                          │   │
│  │   Content-Type: application/json                  │   │
│  │   Cookie: EIPGW-TOKEN=<jwt>                      │   │
│  │ (EIP 网关自动注入 X-EIPGW-USERID)                  │   │
│  │ Body: { events: [...] }                           │   │
│  └──────────────────────────────────────────────────┘   │
└─────────────────────────────────────────────────────────┘
```

## 二、数据模型

新增文件：`packages/shared/src/types/observability.ts`

```typescript
export type ObservabilityEventType =
  | 'user_login' | 'user_logout' | 'chat_question' | 'agent_question'
  | 'error' | 'upgrade_check'

export interface ObservabilityEventItem {
  eventId: string            // 事件唯一标识（UUID），用于服务端去重
  type: ObservabilityEventType
  userId: string             // 用户工号（由 auth-service.getJobId() 填充，服务端以 req.jobId 覆盖）
  userName?: string
  timestamp: number
  question?: string
  questionLength?: number
  modelId?: string
  channelId?: string
  sessionId?: string
  workspaceId?: string
  result?: 'success' | 'failure' | 'pending'
  responseDurationMs?: number
  error?: { type: string; message: string; stack?: string; statusCode?: number; fingerprint?: string }
  breadcrumbs?: Breadcrumb[] // 事件发生前的用户操作路径（error 事件自动附加）
  client: { appVersion: string; platform: string; osVersion: string }
}

export interface Breadcrumb {
  type: 'navigation' | 'user_action' | 'system' | 'http'
  category: string
  message: string
  timestamp: number
  data?: Record<string, unknown>
}

export interface ObservabilityReportBody {
  events: ObservabilityEventItem[]
}

export interface ObservabilityConfig {
  enabled: boolean
  url: string              // 完整上报地址，如 https://host/workmate/observability/events
  timeoutMs?: number
  maxQueueSize?: number
  flushIntervalMs?: number
  maxBatchSize?: number
  sampleRate?: number         // 采样率 0.0~1.0，默认 1.0（全量上报）
  maxEventsPerMinute?: number // 每分钟最大事件数，默认 60
  enableBreadcrumbs?: boolean // 是否启用面包屑，默认 true
}
```

### 3.1 事件入队核心逻辑

```typescript
// observability-service.ts 核心实现概要

import { getToken } from './auth-service'
import { getSettings } from './settings-service'
import { randomUUID } from 'node:crypto'

const breadcrumbs: Breadcrumb[] = []
const MAX_BREADCRUMBS = 20
let eventQueue: ObservabilityEventItem[] = []
let config: ObservabilityConfig | null = null
let flushTimer: ReturnType<typeof setInterval> | null = null

// ===== 初始化 =====

export function init(obsConfig: ObservabilityConfig): void {
  config = obsConfig
  // 启动定时冲刷
  flushTimer = setInterval(() => flushQueue(), config.flushIntervalMs ?? 5000)
}

// ===== 面包屑记录 =====

export function addBreadcrumb(crumb: Omit<Breadcrumb, 'timestamp'>): void {
  if (!config?.enableBreadcrumbs) return
  breadcrumbs.push({ ...crumb, timestamp: Date.now() } as Breadcrumb)
  if (breadcrumbs.length > MAX_BREADCRUMBS) {
    breadcrumbs.shift()
  }
}

// ===== 通用事件入队（含采样/限流） =====

export function reportEvent(item: Omit<ObservabilityEventItem, 'eventId'>): void {
  if (!config?.enabled) return

  const event: ObservabilityEventItem = {
    ...item,
    eventId: randomUUID(),
  } as ObservabilityEventItem

  eventQueue.push(event)
  if (eventQueue.length >= (config.maxQueueSize ?? 200)) {
    flushQueue()
  }
}

// ===== Chat 提问上报 =====

export function reportChatEvent(params: {
  userId: string; userName: string; question: string; modelId: string;
  result: 'success' | 'failure'; responseDurationMs: number; error?: Error;
}): void {
  reportEvent({
    type: 'chat_question',
    userId: params.userId,
    userName: params.userName,
    timestamp: Date.now(),
    question: params.question,
    questionLength: params.question.length,
    modelId: params.modelId,
    result: params.result,
    responseDurationMs: params.responseDurationMs,
    error: params.error ? {
      type: params.error.name,
      message: params.error.message,
      stack: params.error.stack,
    } : undefined,
    client: getClientInfo(),
  })
}

// ===== 登录事件上报 =====

export function reportLoginEvent(
  jobId: string, name: string, result: 'success' | 'failure', error?: Error,
): void {
  reportEvent({
    type: 'user_login',
    userId: jobId,
    userName: name,
    timestamp: Date.now(),
    result,
    error: error ? { type: error.name, message: error.message } : undefined,
    client: getClientInfo(),
  })
}

// ===== 冲刷上报 =====

export async function flushQueue(): Promise<void> {
  if (eventQueue.length === 0) return

  const batch = eventQueue.splice(0, config?.maxBatchSize ?? 50)
  const token = getToken()

  try {
    const response = await fetch(config!.url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        ...(token ? { 'Cookie': `EIPGW-TOKEN=${token}` } : {}),
      },
      body: JSON.stringify({ events: batch }),
      signal: AbortSignal.timeout(config?.timeoutMs ?? 5000),
    })

    if (!response.ok) {
      throw new Error(`HTTP ${response.status}`)
    }
  } catch (error) {
    // 失败时写回队列或本地磁盘缓存
    console.warn('[观测上报] 上报失败:', error)
  }
}

// ===== 关闭 =====

export function shutdown(): void {
  if (flushTimer) clearInterval(flushTimer)
  flushQueue() // 最终冲刷
}
```

## 四、异常捕获与上报

### 4.1 集成点

| 层 | 方式 |
|----|------|
| 主进程 | `process.on('uncaughtException')` / `process.on('unhandledRejection')` |
| 渲染进程 | `ErrorBoundary` + `file-logger.ts` error 级别日志上报 |
| Chat | `chat-service.ts` catch 块中追加 `reportErrorEvent()` |
| Agent | `agent-orchestrator.ts` catch 块中追加 `reportErrorEvent()` |
| 升级检测 | `upgrade-check-service.ts` catch 块中追加 `reportErrorEvent()` |

### 4.2 防递归保护

```typescript
let isReportingError = false

export function reportErrorEvent(error: Error, context?: Record<string, unknown>): void {
  if (isReportingError) return
  isReportingError = true
  try {
    const fingerprint = generateFingerprint(error.name, error.message)
    reportEvent({
      type: 'error',
      userId: getJobId() ?? 'unknown',
      userName: 'system',
      timestamp: Date.now(),
      result: 'failure',
      error: {
        type: error.name,
        message: error.message,
        stack: error.stack,
        fingerprint,
      },
      breadcrumbs: [...breadcrumbs],       // 自动附加最近操作路径
      ...context,
      client: getClientInfo(),
    })
  } finally { isReportingError = false }
}

function generateFingerprint(errorType: string, errorMessage: string): string {
  const raw = `${errorType}:${errorMessage.slice(0, 100)}`
  return createHash('md5').update(raw).digest('hex').slice(0, 16)
}
```

### 4.3 主进程全局异常捕获

新增文件：`apps/electron/src/main/lib/error-handler.ts`

```typescript
import { reportErrorEvent } from './observability-service'
import { app } from 'electron'

export function registerGlobalErrorHandlers(): void {
  process.on('uncaughtException', (error) => {
    console.error('[Global Error] 未捕获异常:', error)
    reportErrorEvent(error, { context: 'uncaughtException' })
  })

  process.on('unhandledRejection', (reason) => {
    console.error('[Global Error] 未处理的 Promise 拒绝:', reason)
    const error = reason instanceof Error ? reason : new Error(String(reason))
    reportErrorEvent(error, { context: 'unhandledRejection' })
  })
}
```

### 4.4 渲染进程 ErrorBoundary

新增文件：`apps/electron/src/renderer/components/ErrorBoundary.tsx`

```typescript
import React from 'react'

interface Props { children: React.ReactNode; fallback?: React.ReactNode }
interface State { hasError: boolean; error?: Error }

export class ErrorBoundary extends React.Component<Props, State> {
  state: State = { hasError: false }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error }
  }

  componentDidCatch(error: Error, errorInfo: React.ErrorInfo): void {
    window.electronAPI?.reportError?.({
      name: error.name,
      message: error.message,
      stack: error.stack,
      componentStack: errorInfo.componentStack,
    })
  }

  render() {
    if (this.state.hasError) {
      return this.props.fallback ?? <div>页面发生错误，请刷新重试</div>
    }
    return this.props.children
  }
}
```

## 六、安全注意事项

| 项 | 措施 |
|----|------|
| 身份不可伪造 | 服务端入库时用 `req.jobId` 覆盖客户端上报的 `userId`，确保身份真实 |
| 上报内容脱敏 | 异常堆栈中过滤本地文件路径中的用户名 |
| 防递归上报 | `isReportingError` 标记防止上报异常触发自身递归 |
| 上报失败 | fire-and-forget，失败写入本地磁盘缓存 |
| 事件去重 | 每事件携带 `eventId`（UUID），服务端幂等去重 |
| 采样与限流 | 非 error 事件受采样率/限流控制，error 事件全量上报 |
