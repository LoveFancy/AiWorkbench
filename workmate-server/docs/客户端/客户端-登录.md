## 一、设计思路

登录不依赖后端 WorkMate Server，而是通过企业 **EIP 网关** 完成认证。客户端只需携带 Cookie 请求，EIP 网关会自动在 Header 中注入加密的工号给后端。

### 核心原则：最小侵入

为降低与 proma 原有代码的冲突风险，所有登录相关代码集中在 `apps/electron/src/auth/` **独立目录**内，原有文件仅做最少量的**接线式修改**（通常每个文件只加 1~3 行）。

```
原有文件修改策略：
  不改逻辑，只加一行 import + 一行调用
```

### 整体流程

```
1. 用户在登录页输入工号 + 密码
          ↓
2. 客户端 POST http://eip.htsc.com.cn/gateway/login
   Body: { "username": "022480", "password": "123123" }
          ↓
3. EIP 网关返回 200:
   Set-Cookie: EIPGW-TOKEN=<short_term_jwt>   ← 短期 Token，有效期 4 小时
          ↓
4. 客户端携带 EIPGW-TOKEN Cookie，请求长期 Token:
   GET http://eip.htsc.com.cn/gateway/manage/user/token/generate?days=365
          ↓
5. EIP 网关返回长期 Token（字符串格式）:
   EIPGW-TOKEN:您的token为：eyJ0eXAiOiJKV1QiLCJhbGciOiJIUzI1NiJ9...
   有效期 365天
          ↓
6. 客户端安全存储：长期 EIPGW-TOKEN + 短期 EIPGW-TOKEN（Set-Cookie 下发） + 签发时间 + 过期时间 + 工号
          ↓
7. 所有后续请求经 EIP 网关，网关自动：
   - 携带 Cookie（EIPGW-TOKEN）
   - 注入 Header X-EIPGW-USERID（AES-128-GCM 加密工号）
          ↓
8. 每次启动 / 定时检查：若自签发日起已过 180 天 → 强制跳转登录页重新认证
```

### 3.1 `types.ts` — 类型定义

```typescript
/** 登录状态 */
export interface AuthState {
  isLoggedIn: boolean
  jobId?: string
  displayName?: string
  lastLoginAt?: number
}

/** 持久化的认证数据 */
export interface PersistedAuthData {
  // —— 长期 Token（用于日常接口调用，有效期 365 天）——

  shortTokenExpiresAt: number   // 短期 Token 过期时间（签发时间 + 4 小时）

  displayName?: string

  tokenExpiresAt?: number

  token: string             // 长期 EIPGW-TOKEN JWT 字符串
  shortToken?: string       // 短期 EIPGW-TOKEN JWT 字符串（可能已过期）
  jobId: string             // 工号
  displayName?: string      // 显示名称
  lastLoginAt: number       // 最后登录时间（Unix 毫秒）
  expiresAt: number         // 长期 Token 自身过期时间（Unix 毫秒）
```

### 3.2 `auth-service.ts` — 认证服务（主进程）

```
函数列表（按调用顺序排列）：
┌── loginWithEipGateway(username, password, days)   — ★ 总入口：完整登录流程
│   ├── login(username, password)                   — Step 1: EIP 网关登录
│   ├── getLongTermToken(shortToken, days)          — Step 2: 获取长期 Token
│   └── saveToken(longTerm, shortTerm, jobId, days) — Step 3: 安全存储（长期 + 短期）
├── getToken(): string | null                       — 读取长期 Token
├── getShortToken(): string | null                  — 读取短期 Token（Set-Cookie 下发）
├── getJobId(): string | null                       — 读取工号
├── isLoggedIn(): boolean                           — 是否已登录
├── needsReauth(): boolean                          — 是否超过 180 天需强制重登
├── getAuthInfo(): AuthInfo | null                  — ★ 统一读取：长期/短期 token + 工号 + 登录时间 + 过期时间 + 是否需要重登
└── logout(): void                                  — 清除登录状态
```

```typescript
// apps/electron/src/auth/auth-service.ts

import { safeStorage } from 'electron'
import { readFileSync, writeFileSync, existsSync } from 'node:fs'
import { join } from 'node:path'
import type { LoginResult, PersistedAuthData } from './types'

// ---- 注意：getConfigDir() 来自原有代码，路径需根据实际项目调整 ----
// import { getConfigDir } from '../main/lib/config-paths'

const AUTH_FILE = 'auth.json'
const EIP_GATEWAY_BASE = 'http://eip.htsc.com.cn/gateway'

/** 强制重新登录天数：自 Token 初始签发起超过此天数必须重新登录 */
const FORCED_REAUTH_DAYS = 180

function getAuthFilePath(configDir: string): string {
  return join(configDir, AUTH_FILE)
}

/** 设置配置目录（由外部注入，避免直接依赖原有代码） */
let _configDir: string | null = null

export function initAuthService(configDir: string): void {
  _configDir = configDir
}

function authPath(): string {
  if (!_configDir) throw new Error('Auth service not initialized: call initAuthService() first')
  return getAuthFilePath(_configDir)
}

// ===== 总入口：完整登录流程 =====

export async function loginWithEipGateway(
  username: string, password: string, longTermDays: number = 365,
): Promise<LoginResult> {
  // Step 1: EIP 网关登录
  const loginResult = await login(username, password)
  if (!loginResult.success || !loginResult.jobId || !loginResult.shortToken) {
    return { success: false, message: loginResult.message }
  }

  // Step 2: 换取长期 Token
  const longTerm = await getLongTermToken(loginResult.shortToken, longTermDays)
  if (!longTerm) {
    return { success: false, message: '获取长期 Token 失败' }
  }

  // Step 3: 安全存储长期 + 短期 Token
  saveToken(longTerm.token, loginResult.shortToken, loginResult.jobId, longTermDays)

  return {
    success: true,
    message: '登录成功',
    jobId: loginResult.jobId,
    tokenExpiresAt: Date.now() + longTermDays * 24 * 60 * 60 * 1000,
  }
}

// ===== Step 1: EIP 网关登录 =====

async function login(
  username: string, password: string,
): Promise<{ success: boolean; message: string; jobId?: string; shortToken?: string }> {
  try {
    const response = await fetch(`${EIP_GATEWAY_BASE}/login`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Origin': 'http://eip.htsc.com.cn',
        'Host': 'eip.htsc.com.cn',
      },
      body: JSON.stringify({ username, password }),
      redirect: 'manual',
    })

    if (response.status !== 200) {
      return { success: false, message: `登录失败: HTTP ${response.status}` }
    }

    const setCookie = response.headers.get('set-cookie') ?? ''
    const token = extractToken(setCookie)
    if (!token) {
      return { success: false, message: '登录失败: 未获取到 EIPGW-TOKEN' }
    }

    const jobId = parseJobId(token)
    return { success: true, message: '登录成功', jobId: jobId ?? username, shortToken: token }
  } catch (error) {
    return { success: false, message: `登录请求异常: ${(error as Error).message}` }
  }
}

// ===== Step 2: 获取长期 Token =====

async function getLongTermToken(
  shortTermToken: string, days: number = 365,
): Promise<{ token: string; days: number } | null> {
  try {
    const response = await fetch(
      `${EIP_GATEWAY_BASE}/manage/user/token/generate?days=${days}`,
      { headers: { 'Cookie': `EIPGW-TOKEN=${shortTermToken}` } },
    )
    if (!response.ok) return null

    const text = await response.text()
    const match = text.match(/EIPGW-TOKEN[：:]\s*您的token为[：:]\s*([A-Za-z0-9\-_.]+)/)
    if (!match) return null
    return { token: match[1], days }
  } catch {
    return null
  }
}

// ===== Step 3: 本地安全存储 =====

function saveToken(longTermToken: string, shortTermToken: string, jobId: string, days: number): void {
  const now = Date.now()
  const expiresAt = now + days * 24 * 60 * 60 * 1000
  // 短期 Token 有效期 4 小时
  const shortTokenExpiresAt = now + 4 * 60 * 60 * 1000
  const authData: PersistedAuthData = {
    jobId,
    expiresAt,
    createdAt: now,
    lastLoginAt: now,
    shortTokenExpiresAt,
  }

  if (safeStorage.isEncryptionAvailable()) {
    authData.encryptedToken = safeStorage.encryptString(longTermToken).toString('base64')
    authData.encryptedShortToken = safeStorage.encryptString(shortTermToken).toString('base64')
  } else {
    authData.token = longTermToken
    authData.shortToken = shortTermToken
  }

  writeFileSync(authPath(), JSON.stringify(authData, null, 2), 'utf-8')
}

// ===== 读取方法（供其他模块调用）=====

export function getToken(): string | null {
  const filePath = authPath()
  if (!existsSync(filePath)) return null

  try {
    const data: PersistedAuthData = JSON.parse(readFileSync(filePath, 'utf-8'))
    if (data.expiresAt && Date.now() > data.expiresAt) return null

    if (data.encryptedToken && safeStorage.isEncryptionAvailable()) {
      return safeStorage.decryptString(Buffer.from(data.encryptedToken, 'base64'))
    }
    return data.token ?? null
  } catch { return null }
}

/** 获取短期 Token（EIP 网关直接下发，有效期 4 小时），用于需要短期凭证的场景 */
export function getShortToken(): string | null {
  const filePath = authPath()
  if (!existsSync(filePath)) return null

  try {
    const data: PersistedAuthData = JSON.parse(readFileSync(filePath, 'utf-8'))
    // 短期 Token 已过期
    if (data.shortTokenExpiresAt && Date.now() > data.shortTokenExpiresAt) return null

    if (data.encryptedShortToken && safeStorage.isEncryptionAvailable()) {
      return safeStorage.decryptString(Buffer.from(data.encryptedShortToken, 'base64'))
    }
    return data.shortToken ?? null
  } catch { return null }
}

export function getJobId(): string | null {
  try {
    const data: PersistedAuthData = JSON.parse(readFileSync(authPath(), 'utf-8'))
    return data.jobId ?? null
  } catch { return null }
}

export function isLoggedIn(): boolean {
  const token = getToken()
  if (!token) return false
  return parseJobId(token) !== null
}

/** 统一对外接口：一次性读取 token + 工号 + 登录时间 + 过期时间 + 是否需要重登 */
export function getAuthInfo(): AuthInfo | null {
  const filePath = authPath()
  if (!existsSync(filePath)) return null

  try {
    const data: PersistedAuthData = JSON.parse(readFileSync(filePath, 'utf-8'))

    let token: string | null = null
    if (data.encryptedToken && safeStorage.isEncryptionAvailable()) {
      token = safeStorage.decryptString(Buffer.from(data.encryptedToken, 'base64'))
    } else if (data.token) {
      token = data.token
    }

    // 长期 Token 自身已过期
    if (!token || (data.expiresAt && Date.now() > data.expiresAt)) {
      return null
    }

    // 读取短期 Token（过期则返回 undefined）
    let shortToken: string | undefined
    if (data.shortTokenExpiresAt && Date.now() <= data.shortTokenExpiresAt) {
      if (data.encryptedShortToken && safeStorage.isEncryptionAvailable()) {
        shortToken = safeStorage.decryptString(Buffer.from(data.encryptedShortToken, 'base64'))
      } else if (data.shortToken) {
        shortToken = data.shortToken
      }
    }

    const now = Date.now()
    const createdAt = data.createdAt ?? data.lastLoginAt  // 兼容旧数据（无 createdAt 字段）
    const needsReauth = (now - createdAt) > FORCED_REAUTH_DAYS * 24 * 60 * 60 * 1000

    return {
      token,
      shortToken,
      jobId: data.jobId,
      displayName: data.displayName,
      lastLoginAt: data.lastLoginAt,
      expiresAt: data.expiresAt,
      shortTokenExpiresAt: data.shortTokenExpiresAt,
      createdAt,
      needsReauth,
    }
  } catch { return null }
}

/** 判断是否超过 180 天需要强制重新登录（不读取 Token 解密，轻量检查） */
export function needsReauth(): boolean {
  const filePath = authPath()
  if (!existsSync(filePath)) return false

  try {
    const data: PersistedAuthData = JSON.parse(readFileSync(filePath, 'utf-8'))
    const createdAt = data.createdAt ?? data.lastLoginAt
    return (Date.now() - createdAt) > FORCED_REAUTH_DAYS * 24 * 60 * 60 * 1000
  } catch { return false }
}

export function logout(): void {
  writeFileSync(authPath(), JSON.stringify({}), 'utf-8')
}

// ===== Cookie 构建（供其他模块如上报/升级检测调用）=====

export function buildAuthHeaders(): Record<string, string> {
  const token = getToken()
  if (token) {
    return { 'Cookie': `EIPGW-TOKEN=${token}` }
  }
  return {}
}

// ===== 工具函数 =====

function extractToken(header: string): string | null {
  const match = header.match(/EIPGW-TOKEN=([^;]+)/)
  return match ? match[1] : null
}

function parseJobId(jwt: string): string | null {
  try {
    const parts = jwt.split('.')
    if (parts.length !== 3) return null
    const payload = JSON.parse(Buffer.from(parts[1], 'base64').toString('utf-8'))
    return payload.mid ?? null
  } catch { return null }
}
```

### 3.3 `ipc-handlers.ts` — IPC 处理器注册

**不修改原有的 `main/ipc.ts`**，而是导出一个注册函数，由 `main/index.ts` 调用。

```typescript
// apps/electron/src/auth/ipc-handlers.ts

import { ipcMain } from 'electron'
import { loginWithEipGateway, getJobId, isLoggedIn, logout } from './auth-service'

const CHANNELS = {
  GET_AUTH_STATE: 'auth:get-state',
  LOGIN: 'auth:login',
  LOGOUT: 'auth:logout',
} as const

/**
 * 注册 auth 相关的 IPC handler。
 * 由 main/index.ts 在 app.whenReady() 中调用。
 *
 * 调用方式（main/index.ts 中加一行）：
 *   import { registerAuthIpcHandlers } from '../auth'
 *   registerAuthIpcHandlers()
 */
export function registerAuthIpcHandlers(): void {
  ipcMain.handle(CHANNELS.GET_AUTH_STATE, () => {
    return { isLoggedIn: isLoggedIn(), jobId: getJobId() }
  })

  ipcMain.handle(CHANNELS.LOGIN, async (_event, username: string, password: string, days?: number) => {
    return loginWithEipGateway(username, password, days)
  })

  ipcMain.handle(CHANNELS.LOGOUT, () => {
    logout()
    return { success: true }
  })
}

export { CHANNELS as AUTH_IPC_CHANNELS }
```

### 3.4 `preload-bridge.ts` — Preload 桥接注册

同样以注册函数形式提供，不修改原有的 preload 主逻辑。

```typescript
// apps/electron/src/auth/preload-bridge.ts

import { contextBridge, ipcRenderer } from 'electron'
import { AUTH_IPC_CHANNELS } from './ipc-handlers'

export interface AuthElectronAPI {
  getAuthState: () => Promise<{ isLoggedIn: boolean; jobId?: string }>
  login: (username: string, password: string, days?: number) => Promise<{
    success: boolean; message: string; jobId?: string; tokenExpiresAt?: number
  }>
  logout: () => Promise<{ success: boolean }>
}

/**
 * 将 auth API 挂载到 window.electronAPI.auth 上。
 *
 * 调用方式（preload/index.ts 中加一行）：
 *   import { registerAuthPreloadBridge } from '../auth'
 *   registerAuthPreloadBridge()
 *
 * 注意：需要确保原有的 contextBridge.exposeInMainWorld('electronAPI', ...)
 * 已预先挂载了 electronAPI 对象，或者在 preload 中调整挂载方式。
 *
 * 推荐做法：在原有的 preload 中留一个空的 electronAPI 壳子，
 * 各模块通过 registerXxxPreloadBridge() 往上面追加属性。
 */
export function registerAuthPreloadBridge(): void {
  // 方式 A：挂载为 electronAPI 的子属性 auth
  // 前提：contextBridge.exposeInMainWorld('electronAPI', { auth: {} }) 已存在
  contextBridge.exposeInMainWorld('electronAPI', {
    auth: {
      getAuthState: () => ipcRenderer.invoke(AUTH_IPC_CHANNELS.GET_AUTH_STATE),
      login: (username: string, password: string, days?: number) =>
        ipcRenderer.invoke(AUTH_IPC_CHANNELS.LOGIN, username, password, days),
      logout: () => ipcRenderer.invoke(AUTH_IPC_CHANNELS.LOGOUT),
    } satisfies AuthElectronAPI,
  })
}
```

#### 3.6.3 各状态 UI 差异

| 状态 | 工号输入框 | 密码输入框 | 登录按钮 | 跳过链接 | 错误提示 |
|------|-----------|-----------|---------|---------|---------|
| IDLE | 空，可输入 | 空，可输入 | 禁用（灰色） | 可见 | 无 |
| READY | 已填，可输入 | 已填，可输入 | 可点击（蓝色） | 可见 | 无 |
| LOADING | 禁用 | 禁用 | loading 动画 + "登录中..." | 隐藏 | 无 |
| ERROR | 保留输入，可编辑 | 清空 | 可点击 | 可见 | 红色文字，如"登录失败：工号或密码错误" |
| SUCCESS | — | — | — | — | 无（自动跳转主界面） |

#### 3.6.4 交互细节

| 交互 | 行为 |
|------|------|
| 工号/密码任一为空 | 登录按钮置灰不可点击 |
| 回车键 | 在密码输入框按 Enter 触发登录 |
| 登录中 | 按钮显示 spinner + "登录中..."，输入框禁用，跳过链接隐藏 |
| 登录失败 | 密码框清空并自动聚焦，顶部显示红色错误信息，3 秒后自动消失 |
| 登录成功 | 更新 `authStateAtom`，1 秒后自动跳转主界面 |
| 跳过登录 | 直接进入主界面，`authStateAtom.isLoggedIn = false`，部分功能受限 |
| 180 天强制重登 | 启动时检测 `needsReauth() === true`，不显示"跳过登录"链接 |

#### 3.6.5 组件结构伪代码

```tsx
// apps/electron/src/auth/LoginView.tsx

import { useState } from 'react'
import { useSetAtom } from 'jotai'
import { authStateAtom } from './atoms'

type LoginState = 'idle' | 'ready' | 'loading' | 'error' | 'success'

export function LoginView({ onLoginSuccess, allowSkip = true }: {
  onLoginSuccess: () => void
  allowSkip?: boolean
}) {
  const [jobId, setJobId] = useState('')
  const [password, setPassword] = useState('')
  const [state, setState] = useState<LoginState>('idle')
  const [errorMsg, setErrorMsg] = useState('')
  const setAuth = useSetAtom(authStateAtom)

  const canSubmit = jobId.trim() !== '' && password.trim() !== ''

  // 输入变化 → 更新状态
  const handleJobIdChange = (value: string) => {
    setJobId(value)
    setState(value && password ? 'ready' : 'idle')
    setErrorMsg('')
  }

  const handlePasswordChange = (value: string) => {
    setPassword(value)
    setState(jobId && value ? 'ready' : 'idle')
    setErrorMsg('')
  }

  // 提交登录
  const handleLogin = async () => {
    if (!canSubmit) return
    setState('loading')
    try {
      const result = await window.electronAPI.auth.login(jobId, password)
      if (result.success) {
        setState('success')
        setAuth({ isLoggedIn: true, jobId: result.jobId })
        setTimeout(onLoginSuccess, 1000)
      } else {
        setState('error')
        setErrorMsg(result.message)
        setPassword('')
      }
    } catch {
      setState('error')
      setErrorMsg('网络异常，请重试')
      setPassword('')
    }
  }

  const handleSkip = () => {
    setAuth({ isLoggedIn: false })
    setState('idle') // 不设为 skipped，直接回调
    onLoginSuccess()
  }

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter' && canSubmit && state !== 'loading') {
      handleLogin()
    }
  }

  return (
    <div className="login-view">
      {/* Logo + 欢迎文案 */}
      <div className="login-header">
        <img src="./assets/logo.svg" alt="WorkMate" className="logo" />
        <h1>WorkMate AI 研发助手</h1>
        <p>欢迎使用 WorkMate AI 研发助手</p>
        <p className="subtitle">请使用公司工号登录 EIP 网关</p>
      </div>

      {/* 工号输入框 */}
      <div className="input-group">
        <span className="input-icon">👤</span>
        <input
          type="text"
          placeholder="请输入工号"
          value={jobId}
          onChange={(e) => handleJobIdChange(e.target.value)}
          disabled={state === 'loading'}
          autoFocus
        />
      </div>

      {/* 密码输入框 */}
      <div className="input-group">
        <span className="input-icon">🔒</span>
        <input
          type="password"
          placeholder="请输入密码"
          value={password}
          onChange={(e) => handlePasswordChange(e.target.value)}
          onKeyDown={handleKeyDown}
          disabled={state === 'loading'}
        />
      </div>

      {/* 登录按钮 */}
      <button
        className="login-btn"
        disabled={!canSubmit || state === 'loading'}
        onClick={handleLogin}
      >
        {state === 'loading' ? '登录中...' : '登 录'}
      </button>

      {/* 错误提示 */}
      {state === 'error' && (
        <p className="error-msg">{errorMsg}</p>
      )}

      {/* 跳过登录（仅在允许跳过且不是强制重登时显示） */}
      {allowSkip && (
        <a className="skip-link" onClick={handleSkip}>
          跳过登录，直接使用
        </a>
      )}
    </div>
  )
}
```

## 四、Cookie 传递约定

客户端请求经过 EIP 网关时，**网关自动完成以下操作**：

1. 校验请求携带的 Cookie（`EIPGW-TOKEN`）
2. 校验通过后，在 Header 中注入 `X-EIPGW-USERID`（AES-128-GCM 加密的工号）

客户端只需要确保请求携带 Cookie 即可，**无需自行处理工号加密**。其他模块（上报、升级检测等）通过 `buildAuthHeaders()` 获取带 Cookie 的请求头：

```typescript
import { buildAuthHeaders } from '../auth'

// 方式 A：底层手动构建（适用于需要自定义请求头的场景）
fetch(url, {
  headers: {
    'Content-Type': 'application/json',
    ...buildAuthHeaders(),
  },
})

// 方式 B：推荐使用统一 HTTP 工具（自动注入 Cookie）
// 详见 [客户端-华泰EIP调用.md](./客户端-华泰EIP调用.md)
import { httpGet, httpPost } from '../shared/hteip-client'
const { ok, data } = await httpGet<MyResponse>(`${serverUrl}/workmate/models`)
```

```typescript
    expect(info!.needsReauth).toBe(false)  // 刚登录，不超 180 天
    expect(info!.createdAt).toBeGreaterThan(0)
    expect(info!.lastLoginAt).toBeGreaterThan(0)
  })

  it('needsReauth - 未超 180 天返回 false', async () => {
    globalThis.fetch = mockEipGateway()
    const { loginWithEipGateway, needsReauth } = require('../auth-service')
    await loginWithEipGateway('022480', 'pass')

    expect(needsReauth()).toBe(false)
  })

  it('needsReauth - 超过 180 天返回 true', async () => {
    globalThis.fetch = mockEipGateway()
    const { loginWithEipGateway } = require('../auth-service')
    await loginWithEipGateway('022480', 'pass')

    // 手动修改 auth.json 中的 createdAt 为 181 天前
    const authFile = JSON.parse(readFileSync(join(tempDir, 'auth.json'), 'utf-8'))
    authFile.createdAt = Date.now() - 181 * 24 * 60 * 60 * 1000
    writeFileSync(join(tempDir, 'auth.json'), JSON.stringify(authFile))

    const { needsReauth } = require('../auth-service')
    expect(needsReauth()).toBe(true)
  })

  it('getAuthInfo - needsReauth 为 true 时仍返回 token', async () => {
    globalThis.fetch = mockEipGateway()
    const { loginWithEipGateway } = require('../auth-service')
    await loginWithEipGateway('022480', 'pass')

    // 修改为 181 天前
    const authFile = JSON.parse(readFileSync(join(tempDir, 'auth.json'), 'utf-8'))
    authFile.createdAt = Date.now() - 181 * 24 * 60 * 60 * 1000
    writeFileSync(join(tempDir, 'auth.json'), JSON.stringify(authFile))

    const { getAuthInfo } = require('../auth-service')
    const info = getAuthInfo()
    expect(info).not.toBeNull()
    expect(info!.needsReauth).toBe(true)
    expect(info!.token).toBeTruthy()  // Token 仍在，由调用方决定是否跳转登录
  })

  it('Token 自身过期 → getAuthInfo 返回 null', async () => {
    globalThis.fetch = mockEipGateway()
    const { loginWithEipGateway } = require('../auth-service')
    await loginWithEipGateway('022480', 'pass')

    // 手动修改 expiresAt 为已过期
    const authFile = JSON.parse(readFileSync(join(tempDir, 'auth.json'), 'utf-8'))
    authFile.expiresAt = Date.now() - 1000
    writeFileSync(join(tempDir, 'auth.json'), JSON.stringify(authFile))

    const { getAuthInfo } = require('../auth-service')
    expect(getAuthInfo()).toBeNull()
  })
})
```

### 8.10 Mock 要点总结

| 要 mock 的内容 | 原因 | 方法 |
|---------------|------|------|
| `safeStorage` | Electron 原生模块，Node 环境不可用 | `vi.mock('electron', ...)` |
| `fetch` | EIP 网关不可达 | `globalThis.fetch = vi.fn()` |
| `safeStorage.encryptString()` | 需要模拟加密/解密双向 | 内存 Map 存储 |
| 文件系统 | 直接使用真实 `fs` | `mkdtempSync` 创建临时目录 |
| 时间 | `createdAt` / `expiresAt` 判定 | 直接修改 `auth.json` 中的时间戳 |
